3. SYNTAX DML - INSERT DATA (menggunakan Python + psycopg2)

import pandas as pd
import psycopg2
from psycopg2.extras import execute_values

# Koneksi ke PostgreSQL
db_user     = "airflow"
db_password = "airflow"
db_host     = "localhost"
db_port     = "5434"
db_name     = "airflow"

connection = psycopg2.connect(
    user=db_user,
    password=db_password,
    host=db_host,
    port=db_port,
    database=db_name
)
cursor = connection.cursor()

# Load dataset
df = pd.read_csv('P2M3_farhan_hl_data_raw.csv', low_memory=False)

# Insert data ke table_m3
cols         = ",".join([f'"{col}"' for col in df.columns])
values       = [tuple(row) for row in df.itertuples(index=False, name=None)]
insert_query = f"INSERT INTO table_m3 ({cols}) VALUES %s"

cursor.execute("TRUNCATE TABLE table_m3")
execute_values(cursor, insert_query, values)
connection.commit()

# Close connection
cursor.close()
connection.close()

print(f"Data berhasil dimasukkan ke PostgreSQL: {len(values)} rows.")
